s = %s

G1 = 1 / (1 + 0.1*s);
sys1 = syslin('c', G1);

G2 = 1 / (1 + 1*s);
sys2 = syslin('c', G2);

G3 = 1 / (1 + 10*s);
sys3 = syslin('c', G3);

clf(0);
bode([sys1;sys2;sys3], 0.01, 100, ['G1';'G2';'G3']);

xs2png(0, 'fols_bode.png');


clf(0);
nyquist([sys1;sys2;sys3], 0.01, 100, ['G1';'G2';'G3']);

xs2png(0, 'fols_nyquist.png');
